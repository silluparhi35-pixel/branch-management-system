    function switchTab(tabName) {
      // Hide all tabs
      const tabs = document.querySelectorAll('.tab-content');
      tabs.forEach(tab => tab.classList.remove('active'));
      
      // Remove active from all nav buttons
      const navTabs = document.querySelectorAll('.nav-tab');
      navTabs.forEach(tab => tab.classList.remove('active'));
      
      // Show selected tab
      document.getElementById(tabName).classList.add('active');
      
      // Add active to clicked nav button
      event.target.classList.add('active');
      
      // Load data if needed
      if (tabName === 'branches') {
        loadBranches();
      } else if (tabName === 'users') {
        loadUsers();
      } else if (tabName === 'cashoffice') {
        loadCashOffices();
      }

    function submitCreateBranch() {
      const errorDiv = document.getElementById('createErrorMsg');
      errorDiv.style.display = 'none';
      
      const branchData = {
        branchCode: document.getElementById('branchCode').value.trim().toUpperCase(),
        branchName: document.getElementById('branchName').value.trim(),
        pinCode: document.getElementById('pinCode').value.trim(),
        address: document.getElementById('address').value.trim(),
        contact: document.getElementById('contact').value.trim(),
        supervisorId: document.getElementById('supervisorId').value.trim()
      };
      
      // Validate required fields
      if (!branchData.branchCode || !branchData.branchName || !branchData.pinCode) {
        errorDiv.textContent = '❌ Please fill in all required fields!';
        errorDiv.style.display = 'block';
        return;
      }
      
      if (!/^\d{6}$/.test(branchData.pinCode)) {
        errorDiv.textContent = '❌ PIN Code must be exactly 6 digits!';
        errorDiv.style.display = 'block';
        return;
      }
      
      if (branchData.contact && !/^\d{10}$/.test(branchData.contact)) {
        errorDiv.textContent = '❌ Contact number must be 10 digits!';
        errorDiv.style.display = 'block';
        return;
      }
      
      google.script.run
        .withSuccessHandler(function(result) {
          if (result.success) {
            alert('✅ ' + result.message);
            closeCreateBranchModal();
            loadBranches();
          } else {

          errorDiv.style.display = 'block';
        })
        .createBranch(branchData, currentAdminId);
    }

    function updateBranchFilters() {

      const createUserBranchSelect = document.getElementById('createUserBranch');
      
      if (filterBranchSelect) {
        // Clear existing options except first

          filterBranchSelect.remove(1);
        }
        
        // Add active branches
        allBranches.forEach(branch => {
          if (branch.status === 'ACTIVE') {
            const option = document.createElement('option');
            option.value = branch.branchCode;
            option.textContent = branch.branchCode + ' - ' + branch.branchName;
            filterBranchSelect.appendChild(option);
          }
        });
      }
      
      // Update create user branch dropdown
      if (createUserBranchSelect) {
        // Clear existing options except first
        while (createUserBranchSelect.options.length > 1) {
          createUserBranchSelect.remove(1);
        }
        
        // Add active branches
        allBranches.forEach(branch => {
          if (branch.status === 'ACTIVE') {
            const option = document.createElement('option');
            option.value = branch.branchCode;
            option.textContent = branch.branchCode + ' - ' + branch.branchName;
            createUserBranchSelect.appendChild(option);
          }
        });
      }
    }

    function applyFilters() {
      const filterBranchElem = document.getElementById('filterBranch');
      const filterRoleElem = document.getElementById('filterRole');
      const filterStatusElem = document.getElementById('filterStatus');
      
      const branchFilter = filterBranchElem ? filterBranchElem.value : '';
      const roleFilter = filterRoleElem ? filterRoleElem.value : '';
      const statusFilter = filterStatusElem ? filterStatusElem.value : '';
      
      let filtered = allUsers.filter(user => {
        if (branchFilter && user.branchCode !== branchFilter) return false;
        if (roleFilter && user.role !== roleFilter) return false;
        if (statusFilter && user.status !== statusFilter) return false;
        return true;
      });
      
      displayUsers(filtered);
    }

      const filterBranchElem = document.getElementById('filterBranch');

      const filterStatusElem = document.getElementById('filterStatus');

      if (filterRoleElem) filterRoleElem.value = '';

      displayUsers(allUsers);
    }

    function submitCreateUser() {
      const errorDiv = document.getElementById('createUserError');
      errorDiv.style.display = 'none';
      
      const role = document.getElementById('createUserRole').value;
      const isTreasurer = (role === 'TREASURER');
      
      const userData = {
        userId:     document.getElementById('createUserId').value.trim().toUpperCase(),
        name:       document.getElementById('createUserName').value.trim(),
        role:       role,
        branchCode: isTreasurer ? '' : document.getElementById('createUserBranch').value,
        officeId:   isTreasurer ? document.getElementById('createUserOffice').value : '',
        contact:    document.getElementById('createUserContact').value.trim(),
        email:      document.getElementById('createUserEmail').value.trim()
      };
      
      // Validate
      if (!userData.userId || !userData.name || !userData.role) {
        errorDiv.textContent = '❌ Please fill in all required fields (ID, Name, Role)!';
        errorDiv.style.display = 'block';
        return;
      }
      // Branch required for non-treasurer
      if (!isTreasurer && !userData.branchCode) {
        errorDiv.textContent = '❌ Please select a branch!';
        errorDiv.style.display = 'block';
        return;
      }
      
      if (userData.contact && !/^\d{10}$/.test(userData.contact)) {
        errorDiv.textContent = '❌ Contact number must be 10 digits!';
        errorDiv.style.display = 'block';
        return;
      }
      
      // Show loading
      errorDiv.textContent = '⏳ Creating user...';
      errorDiv.style.display = 'block';
      errorDiv.style.background = '#d1ecf1';
      errorDiv.style.color = '#0c5460';
      errorDiv.style.borderLeftColor = '#17a2b8';
      
      google.script.run
        .withSuccessHandler(function(result) {
          if (result.success) {
            // Show success modal with credentials
            const createdUserIdElem = document.getElementById('createdUserId');
            const createdPasswordElem = document.getElementById('createdPassword');
            const createdRoleElem = document.getElementById('createdRole');
            const createdBranchElem = document.getElementById('createdBranch');
            
            if (createdUserIdElem) createdUserIdElem.textContent = result.userId;
            if (createdPasswordElem) createdPasswordElem.textContent = result.defaultPassword;
            if (createdRoleElem) createdRoleElem.textContent = result.role;
            if (createdBranchElem) {
              if (result.role === 'TREASURER') {
                createdBranchElem.textContent = result.officeId ? 'Cash Office: ' + result.officeId : 'No office assigned';
              } else {
                createdBranchElem.textContent = (result.branchCode || '') + (result.branchName ? ' - ' + result.branchName : '');
              }
            }
            
            closeCreateUserModal();
            document.getElementById('successModal').style.display = 'flex';
            
            loadUsers();
            loadUserStatistics();
          } else {
            errorDiv.textContent = '❌ ' + result.message;
            errorDiv.style.display = 'block';
            errorDiv.style.background = '#f8d7da';
            errorDiv.style.color = '#721c24';
            errorDiv.style.borderLeftColor = '#e74c3c';
          }
        })
        .withFailureHandler(function(error) {
          errorDiv.textContent = '❌ Error: ' + error.message;
          errorDiv.style.display = 'block';
          errorDiv.style.background = '#f8d7da';
          errorDiv.style.color = '#721c24';
          errorDiv.style.borderLeftColor = '#e74c3c';
        })
        .createUser(userData, currentAdminId);

      successDiv.style.display = 'none';
      
      const currentPassword = document.getElementById('currentPassword').value;

                  <button class="btn btn-danger" onclick="deleteOfficeConfirm('${o.officeId}')">🗑️ Delete</button>
                </div>
              </div>`;
          });
          listDiv.innerHTML = html;
          
          // Load branch counts + populate treasurer dropdowns
          result.offices.forEach(function(o) {
            loadOfficeBranchCount(o.officeId);
            populateTreasurerDropdown(o.officeId, o.treasurerId);
          });
        })
        .withFailureHandler(function(error) {
          listDiv.innerHTML = `<div class="empty-state"><div class="empty-state-icon">❌</div><h3>Error</h3><p>${error.message}</p></div>`;
        })
        .getAllCashOffices();
    }
    
    function populateTreasurerDropdown(officeId, currentTreasurerId) {
      // Filter allUsers to get TREASURER role only
      const sel = document.getElementById('treasurerSelect_' + officeId);
      if (!sel) return;
      sel.innerHTML = '<option value="">-- Select Treasurer --</option>';
      
      // If allUsers already loaded, use it; else fetch
      const treasurers = allUsers.filter(function(u) {
        return u.role === 'TREASURER' && u.status === 'ACTIVE';
      });
      
      if (treasurers.length > 0) {
        treasurers.forEach(function(t) {
          const opt = document.createElement('option');
          opt.value = t.userId;
          opt.textContent = t.userId + ' — ' + t.name;
          if (currentTreasurerId && t.userId === currentTreasurerId) opt.selected = true;
          sel.appendChild(opt);
        });
      } else {
        // Fetch from backend if not loaded
        google.script.run
          .withSuccessHandler(function(res) {
            if (!res || !res.users) return;
            const trs = res.users.filter(function(u) { return u.role === 'TREASURER' && u.status === 'ACTIVE'; });
            sel.innerHTML = '<option value="">-- Select Treasurer --</option>';
            trs.forEach(function(t) {
              const opt = document.createElement('option');
              opt.value = t.userId;
              opt.textContent = t.userId + ' — ' + t.name;
              if (currentTreasurerId && t.userId === currentTreasurerId) opt.selected = true;
              sel.appendChild(opt);
            });
            if (trs.length === 0) {
              sel.innerHTML = '<option value="">-- No Treasurer users yet --</option>';
            }
          })
          .withFailureHandler(function() {})
          .getAllUsers(currentAdminId);
      }
    }

    function viewOfficeBranches(officeId) {
      google.script.run
        .withSuccessHandler(function(res) {
          if (!res || !res.success) {
            alert('❌ ' + (res ? res.message : 'Error loading branches'));
            return;
          }
          const branches = res.branches || [];
          if (branches.length === 0) {
            alert('ℹ️ Office ' + officeId + ' mein koi branch map nahi hai abhi.\n\n"Map Branch" button se branch add karo.');
          } else {
            const list = branches.map(function(b) {
              const trs = b.treasurerId ? ' [Treasurer: ' + b.treasurerId + ']' : ' [No Treasurer]';
              return '• ' + b.branchCode + (b.branchName ? ' — ' + b.branchName : '') + trs;
            }).join('\n');
            alert('🏦 Branches mapped to office ' + officeId + ':\n\n' + list);
          }
        })
        .withFailureHandler(function(e) {
          alert('❌ Error: ' + e.message);
        })
        .getCashOfficeBranches(officeId);
    }

    function loadMappedBranchesForRemoval(officeId) {
      const listDiv = document.getElementById('removeBranchList');
      listDiv.innerHTML = '<div class="loading" style="display:block;"><div class="spinner"></div><p>Loading...</p></div>';

      google.script.run
        .withSuccessHandler(function(res) {
          if (!res || !res.success) {
            listDiv.innerHTML = '<p style="color:#e74c3c;text-align:center;">❌ ' + (res ? res.message : 'Error') + '</p>';
            return;
          }

          // Filter only ACTIVE branches
          const activeBranches = (res.branches || []).filter(function(b) {
            return b.status === 'ACTIVE';
          });

          if (activeBranches.length === 0) {
            listDiv.innerHTML =
              '<div class="empty-state" style="padding:30px;">' +
              '<div class="empty-state-icon">🏢</div>' +
              '<h3>No Active Branches</h3>' +
              '<p>Is office mein koi active mapped branch nahi hai.</p>' +
              '</div>';
            return;
          }

          // Use DOM approach — no quote escaping issues
          var container = document.createElement('div');
          container.style.cssText = 'display:flex;flex-direction:column;gap:10px;';

          activeBranches.forEach(function(b) {
            var row = document.createElement('div');
            row.style.cssText = 'display:flex;justify-content:space-between;align-items:center;' +
              'background:#f8f9fa;padding:14px 16px;border-radius:10px;border-left:4px solid #3498db;';

            var info = document.createElement('div');
            var codeDiv = document.createElement('div');
            codeDiv.style.cssText = 'font-weight:bold;color:#2c3e50;font-size:15px;';
            codeDiv.textContent = b.branchCode;

            var nameDiv = document.createElement('div');
            nameDiv.style.cssText = 'font-size:12px;color:#7f8c8d;';
            nameDiv.textContent = (b.branchName || '') + (b.mappedDate ? ' · Mapped: ' + b.mappedDate : '');

            info.appendChild(codeDiv);
            info.appendChild(nameDiv);

            var btn = document.createElement('button');
            btn.textContent = '🔴 Remove';
            btn.style.cssText = 'background:linear-gradient(135deg,#e74c3c,#c0392b);color:white;' +
              'border:none;padding:9px 16px;border-radius:8px;font-size:13px;' +
              'font-weight:bold;cursor:pointer;white-space:nowrap;';
            btn.setAttribute('data-branch', b.branchCode);
            btn.addEventListener('click', function() {
              confirmRemoveBranch(this.getAttribute('data-branch'));
            });

            row.appendChild(info);
            row.appendChild(btn);
            container.appendChild(row);
          });

          listDiv.innerHTML = '';
          listDiv.appendChild(container);
        })
        .withFailureHandler(function(err) {
          listDiv.innerHTML = '<p style="color:#e74c3c;text-align:center;">❌ Error: ' + err.message + '</p>';
        })
        .getCashOfficeBranchesAll(officeId, currentAdminId);
    }

    function confirmRemoveBranch(branchCode) {
      if (!confirm('⚠️ Branch ' + branchCode + ' ko office ' + _removeOfficeId + ' se remove karna chahte ho?\n\nIs branch ke pending cash transactions affect ho sakte hain.')) {
        return;
      }

      const errDiv = document.getElementById('removeBranchOfficeError');
      const sucDiv = document.getElementById('removeBranchOfficeSuccess');
      errDiv.style.display = 'none';
      sucDiv.style.display = 'none';

      google.script.run
        .withSuccessHandler(function(res) {
          if (res && res.success) {
            sucDiv.textContent = '✅ ' + res.message;
            sucDiv.style.display = 'block';
            // Reload the branch list
            setTimeout(function() {
              loadMappedBranchesForRemoval(_removeOfficeId);
              loadCashOffices();
            }, 800);
          } else {
            errDiv.textContent = '❌ ' + (res ? res.message : 'Error');
            errDiv.style.display = 'block';
          }
        })
        .withFailureHandler(function(err) {
          errDiv.textContent = '❌ Error: ' + err.message;
          errDiv.style.display = 'block';
        })
        .unmapBranchFromCashOffice(_removeOfficeId, branchCode, currentAdminId);
    }
