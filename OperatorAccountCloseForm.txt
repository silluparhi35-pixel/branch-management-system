  <script>
    // ✅ FIXED INIT: Single clean load event — no race conditions, no dead code
    let operatorId    = null;
    let operatorToken = null;
    let userBranchCode = '';

    window.addEventListener('load', function() {
      // Step 1: Get operatorId — URL param first, then window inject fallback
      operatorId    = new URLSearchParams(window.location.search).get('operator') || (window.OPERATOR_ID || null);
      operatorToken = (new URLSearchParams(window.location.search).get('token'))
                      ? decodeURIComponent(new URLSearchParams(window.location.search).get('token'))
                      : (window.OPERATOR_TOKEN || null);

      if (!operatorId) {
        alert('❌ Operator ID not found!\n\nPlease close this window and login again.');
        window.close();
        return;
      }

      // Step 2: Show operator ID in header
      const header = document.getElementById('headerTitle');
      if (header) {
        header.innerHTML += '<br><small style="font-size:14px;opacity:0.9;">Operator: ' + operatorId + '</small>';
      }

      // Step 3: Fetch branch code + show branch banner
      google.script.run
        .withSuccessHandler(function(result) {
          if (result && result.success) {
            userBranchCode = result.branchCode || '';
            const bannerDiv = document.createElement('div');
            bannerDiv.style.cssText = 'background:linear-gradient(135deg,#667eea,#764ba2);color:white;padding:8px 20px;text-align:center;font-size:13px;';
            bannerDiv.innerHTML = '<strong>Operator:</strong> ' + operatorId + ' &nbsp;|&nbsp; <strong>Branch:</strong> ' + userBranchCode;
            const container = document.querySelector('.container') || document.body;
            container.insertBefore(bannerDiv, container.firstChild);
          }
        })
        .getOperatorDetails(operatorId);

      // Step 4: Load form data
      initFormData();
    });

    function initFormData() {
      // ✅ FIX: Set today's date on load
      document.getElementById('closeDate').valueAsDate = new Date();
      document.getElementById('closeDate').readOnly = true;
      loadActiveAccounts();
    }



    function loadActiveAccounts() {
      google.script.run
        .withSuccessHandler(function(res) {
          if (res.success) {
            const select = document.getElementById('accountNo');
            res.accounts.forEach(acc => {
              const option = document.createElement('option');
              option.value = acc.accountNo;
              option.textContent = acc.accountNo + ' - ' + acc.name;
              option.dataset.name            = acc.name;
              option.dataset.accountType     = acc.accountType;
              option.dataset.hasLoan         = acc.hasLoan;
              option.dataset.loanOutstanding = acc.loanOutstanding || 0;
              option.dataset.userId          = acc.userId || '';
              option.dataset.email           = acc.email  || '';
              select.appendChild(option);
            });
          }
        })
        .getActiveAccountsForClose(operatorId);
    }

    document.getElementById('accountNo').addEventListener('change', function() {
      const selected = this.options[this.selectedIndex];
      if (selected.value) {
        loadAccountDetails(selected.value);
        document.getElementById('userId').value = selected.dataset.userId || '';
        document.getElementById('email').value  = selected.dataset.email  || '';
      } else {
        document.getElementById('accountInfo').style.display = 'none';
        document.getElementById('loanWarning').style.display = 'none';
        document.getElementById('errorMsg').style.display = 'none';
        // Reset button
        const submitBtn = document.getElementById('submitBtn');
        submitBtn.disabled = false;
        submitBtn.textContent = '📤 Submit for Supervisor Approval';
        submitBtn.style.background = '';
        submitBtn.style.cursor = 'pointer';
      }
    });

    function loadAccountDetails(accountNo) {
      google.script.run
        .withSuccessHandler(function(res) {
          if (res.success) {
            const acc = res.account;
            const submitBtn = document.getElementById('submitBtn');
            const errorDiv = document.getElementById('errorMsg');
            const loanWarningDiv = document.getElementById('loanWarning');
            const loanWarningText = document.getElementById('loanWarningText');
            
            document.getElementById('memberName').textContent = acc.name;
            document.getElementById('accountType').textContent = acc.accountType;
            document.getElementById('currentBalance').textContent = acc.currentBalance.toFixed(2);
            document.getElementById('hasLoan').textContent = acc.hasLoan ? 'YES ⚠️' : 'NO ✅';
            document.getElementById('loanOutstanding').textContent = acc.loanOutstanding.toFixed(2);
            
            // Pre-fill settlement amount with current balance
            document.getElementById('settlementAmount').value = acc.currentBalance;
            
            document.getElementById('accountInfo').style.display = 'block';
            
            // ✅ DISABLE SUBMIT if account has active loan
            if (acc.hasLoan || acc.loanOutstanding > 0) {
              submitBtn.disabled = true;
              submitBtn.textContent = '🚫 Cannot Close - Active Loan Exists';
              submitBtn.style.background = '#dc3545';
              submitBtn.style.cursor = 'not-allowed';
              
              loanWarningText.innerHTML = 
                'This account has an <strong>active loan</strong> of <strong>₹' + acc.loanOutstanding.toFixed(2) + '</strong><br><br>' +
                '✋ All loans must be <strong>fully closed</strong> before account closure.<br><br>' +
                '📋 <strong>Action Required:</strong><br>' +
                '1. Close the outstanding loan first<br>' +
                '2. Wait for loan settlement confirmation<br>' +
                '3. Then retry account closure';
              
              loanWarningDiv.style.display = 'block';
              errorDiv.style.display = 'none';
              
              // Disable settlement amount field
              document.getElementById('settlementAmount').disabled = true;
              document.getElementById('closeDate').disabled = true;
            } else {
              submitBtn.disabled = false;
              submitBtn.textContent = '📤 Submit for Supervisor Approval';
              submitBtn.style.background = '';
              submitBtn.style.cursor = 'pointer';
              loanWarningDiv.style.display = 'none';
              errorDiv.style.display = 'none';
              
              // Enable fields
              document.getElementById('settlementAmount').disabled = false;
              document.getElementById('closeDate').disabled = false;
            }
          }
        })
        .getAccountDetailsForClose(accountNo);
    }

    document.getElementById('closeForm').addEventListener('submit', function(e) {
      e.preventDefault();
      
      // Double-check loan status before submission
      const submitBtn = document.getElementById('submitBtn');
      if (submitBtn.disabled) {
        showError('❌ Cannot submit: Account has active loan');
        return;
      }
      
      submitClosure();
    });

    function submitClosure() {
      const btn = document.getElementById('submitBtn');
      const errorDiv = document.getElementById('errorMsg');
      const successDiv = document.getElementById('successMsg');

      errorDiv.style.display = 'none';
      successDiv.style.display = 'none';

      // ✅ CRITICAL: Verify operator ID before submission
      if (!operatorId || operatorId === 'null' || operatorId === 'undefined') {
        showError('❌ SUBMISSION BLOCKED: Operator ID is missing!\n\nPlease close this form and login again.');
        return;
      }

      const closeData = {
        accountNo:        document.getElementById('accountNo').value,
        name:             document.getElementById('memberName').textContent,
        settlementAmount: parseFloat(document.getElementById('settlementAmount').value),
        closeDate:        document.getElementById('closeDate').value,
        email:            document.getElementById('email').value.trim(),
        userId:           document.getElementById('userId').value.trim(),
        branchCode:       userBranchCode
      };

      if (!closeData.accountNo || !closeData.settlementAmount) {
        showError('❌ Please fill required fields');
        return;
      }

      if (closeData.settlementAmount < 0) {
        showError('❌ Settlement amount cannot be negative');
        return;
      }

      btn.disabled = true;
      btn.textContent = '⏳ Submitting...';

      console.log('📤 Submitting account close with Operator ID:', operatorId);
      console.log('📋 Close Data:', closeData);

      google.script.run
        .withSuccessHandler(function(res) {
          btn.disabled = false;
          btn.textContent = '📤 Submit for Supervisor Approval';

          if (res.success) {
            showSuccess(`
              <strong>✅ ${res.message}</strong><br><br>
              <strong>Entry ID:</strong> ${res.entryId}<br>
              <strong>Submitted by:</strong> ${operatorId}<br><br>
              <strong>Status:</strong> Pending supervisor approval
            `);
            document.getElementById('closeForm').reset();
            document.getElementById('closeDate').valueAsDate = new Date();
      document.getElementById('closeDate').readOnly = true; // ✅ No backdate
            document.getElementById('accountInfo').style.display = 'none';
            document.getElementById('loanWarning').style.display = 'none';
          } else {
            showError('❌ ' + res.message);
          }
        })
        .withFailureHandler(function(err) {
          btn.disabled = false;
          btn.textContent = '📤 Submit for Supervisor Approval';
          showError('❌ System Error: ' + err.message + '\n\nOperator ID: ' + operatorId);
        })
        // ✅ FIXED: operatorId PEHLE, closeData BAAD MEIN
        .operatorSubmitAccountClose(operatorId, closeData, operatorToken);
    }

    function showError(msg) {
      document.getElementById('errorMsg').innerHTML = msg;
      document.getElementById('errorMsg').style.display = 'block';
      document.getElementById('successMsg').style.display = 'none';
    }

    function showSuccess(msg) {
      document.getElementById('successMsg').innerHTML = msg;
      document.getElementById('successMsg').style.display = 'block';
      document.getElementById('errorMsg').style.display = 'none';
    }

  </script>
