<script>
    // ✅ XSS Protection
    function escapeHtml(str) {
      if (str === null || str === undefined) return '';
      return String(str)
        .replace(/&/g,'&amp;').replace(/</g,'&lt;')
        .replace(/>/g,'&gt;').replace(/"/g,'&quot;')
        .replace(/'/g,'&#x27;');
    }

  // ── State ──
  let selectedAccount   = null;
  let selectedPrintType = null;
  let transactionsData  = null;
  let manualStartSrNo   = 0;   // operator-set Sr. No. — global so async callback can read it

  // ════════════════════════════════════════════════
  // ✅ FIX: currentUser — window.onload ke andar set
  //    karo taaki window.OPERATOR_ID inject ho sake
  //    GAS se pehle
  // ════════════════════════════════════════════════
  let currentUser = 'OPERATOR'; // default

  window.onload = function() {
    // ✅ window.onload ke baad read karo — GAS injection complete hoti hai
    currentUser =
      window.OPERATOR_ID ||
      new URLSearchParams(window.location.search).get('operator') ||
      new URLSearchParams(window.location.search).get('operatorId') ||
      'OPERATOR';

    // ✅ Header badge update karo
    const badge = document.getElementById('operatorBadge');
    if (badge) {
      badge.textContent = currentUser !== 'OPERATOR'
        ? '🔒 Operator: ' + currentUser
        : '🔒 OPERATOR';
    }

    document.getElementById('headerSubtitle').textContent =
      'Print new passbook or update existing passbook | By: ' + currentUser;

    document.getElementById('searchInput').focus();

    console.log('✅ PassbookPrint loaded | Operator:', currentUser);
  };

  // ── Helpers ──
  function showMessage(msg, type) {
    const b = document.getElementById('messageBox');
    b.className = 'message message-' + type;
    b.innerHTML = msg;
    b.style.display = 'block';
    if (type !== 'error') setTimeout(() => b.style.display = 'none', 5000);
  }

  function showLoading(id) {
    const el = document.getElementById(id);
    el.innerHTML = '<div class="loading"><div class="spinner"></div><p>Loading...</p></div>';
    el.style.display = 'block';
  }

  // ── Sr. No. Input functions ───────────────────────────────────
  function showSrNoInput() {
    document.getElementById('srNoBox').style.display = 'block';

    // System se auto printedCount fetch karo — ek hint ke roop mein
    if (selectedAccount) {
      google.script.run
        .withSuccessHandler(function(res) {
          const infoEl = document.getElementById('srNoAutoInfo');
          if (res && res.success) {
            const pc = res.printedCount || 0;
            infoEl.style.display = 'block';
            if (pc === 0) {
              infoEl.innerHTML = '💡 System record: Is account ka <strong>koi transaction abhi tak print nahi hua</strong>. Agar yeh sahi hai to 0 rakho.';
            } else {
              infoEl.innerHTML = '💡 System record ke mutabik <strong>' + pc + ' transactions</strong> pehle print ho chuke hain. Physical passbook dekh ke confirm karo.';
              // Auto-fill with system value as suggestion
              const inp = document.getElementById('lastSrNoInput');
              if (!inp.dataset.userEdited) {
                inp.value = pc;
                onSrNoChange();
              }
            }
          }
        })
        .withFailureHandler(function() {})
        .getPassbookTransactions(selectedAccount.accountNumber, 'SUBSEQUENT');
    }

    onSrNoChange();
    document.getElementById('lastSrNoInput').focus();
    document.getElementById('srNoBox').scrollIntoView({ behavior: 'smooth' });
  }

  function hideSrNoInput() {
    document.getElementById('srNoBox').style.display = 'none';
  }

  function onSrNoChange() {
    const val  = parseInt(document.getElementById('lastSrNoInput').value) || 0;
    const next = val + 1;
    document.getElementById('nextSrNoDisplay').textContent = 'Sr. No. ' + next;
    document.getElementById('nextSrNoNote').textContent    = val === 0
      ? ' se (fresh start)' : ' se (last printed: ' + val + ')';
    // Mark as user-edited so auto-fill doesn't override
    document.getElementById('lastSrNoInput').dataset.userEdited = '1';
  }

  function loadTransactionsWithSrNo() {
    // ✅ Store in global — async callback mein bhi readable rahega
    manualStartSrNo = parseInt(document.getElementById('lastSrNoInput').value) || 0;
    hideSrNoInput();
    selectedPrintType = 'SUBSEQUENT';
    showMessage('Loading transactions from Sr. No. ' + (manualStartSrNo + 1) + '...', 'info');
    showLoading('previewContent');
    document.getElementById('transactionsPreview').style.display = 'block';

    google.script.run
      .withSuccessHandler(function(res) {
        if (!res) {
          showMessage('❌ Server returned no response. Try again.', 'error');
          document.getElementById('transactionsPreview').style.display = 'none';
          return;
        }
        displayTransactions(res);
      })
      .withFailureHandler(function(err) {
        showMessage('❌ Error: ' + (err.message || err), 'error');
        document.getElementById('transactionsPreview').style.display = 'none';
      })
      .getPassbookTransactions(selectedAccount.accountNumber, 'SUBSEQUENT');
  }

  // ── Search ──
  function searchAccount() {
    const term = document.getElementById('searchInput').value.trim();
    if (!term) { showMessage('Please enter account number, member ID, or name', 'error'); return; }
    showLoading('searchResults');
    google.script.run
      .withSuccessHandler(function(res) {
        if (!res) { showMessage('No response from server. Try again.', 'error'); return; }
        displaySearchResults(res);
      })
      .withFailureHandler(function(err) {
        showMessage('Error: ' + (err.message || err), 'error');
        document.getElementById('searchResults').innerHTML = '';
      })
      .searchAccountForPassbook(term);
  }

  function displaySearchResults(res) {
    const div = document.getElementById('searchResults');
    if (!res.success || !res.results || res.results.length === 0) {
      div.innerHTML = '<p style="color:#e74c3c;padding:15px;">❌ No accounts found.</p>';
      div.style.display = 'block';
      return;
    }
    let html = '<h4 style="margin:10px 0;color:#2c3e50;">Select Account:</h4>';
    res.results.forEach(function(acc) {
      html += '<div class="account-card" onclick=\'selectAccount(' +
        JSON.stringify(acc).replace(/'/g, "&#39;") + ')\'>' +
        '<h4>A/C: ' + escapeHtml(acc.accountNumber) + ' — ' + escapeHtml(acc.memberName) + '</h4>' +
        '<p>Scheme: ' + escapeHtml(acc.schemeName) + ' | Member ID: ' + escapeHtml(acc.memberId) +
        ' | Balance: ₹' + (acc.balance || 0).toFixed(2) + '</p>' +
        '</div>';
    });
    div.innerHTML = html;
    div.style.display = 'block';
  }

  function selectAccount(acc) {
    selectedAccount = acc;
    document.getElementById('searchResults').style.display = 'none';
    document.getElementById('printTypeSection').style.display = 'block';
    document.getElementById('printTypeSection').scrollIntoView({ behavior: 'smooth' });
    showMessage('✅ Account selected: ' + acc.accountNumber + ' — ' + acc.memberName, 'success');
  }

  // ── Load Transactions ──
  function loadTransactions(type) {
    selectedPrintType = type;
    showMessage('Loading transactions...', 'info');
    showLoading('previewContent');
    document.getElementById('transactionsPreview').style.display = 'block';

    google.script.run
      .withSuccessHandler(function(res) {
        if (!res) {
          showMessage('❌ Server returned no response. Try again.', 'error');
          document.getElementById('transactionsPreview').style.display = 'none';
          return;
        }
        displayTransactions(res);
      })
      .withFailureHandler(function(err) {
        showMessage('❌ Error: ' + (err.message || err), 'error');
        document.getElementById('transactionsPreview').style.display = 'none';
      })
      .getPassbookTransactions(selectedAccount.accountNumber, type);
  }

  // ── Display & Build Print View ──
  function displayTransactions(res) {
    if (!res.success) {
      showMessage('❌ ' + res.message, 'error');
      document.getElementById('transactionsPreview').style.display = 'none';
      return;
    }

    transactionsData = res;
    const acc = res.accountInfo;

    // Screen preview
    let html = '<div style="background:#e8f5e9;padding:15px;border-radius:8px;margin-bottom:15px;border-left:5px solid #27ae60;">' +
      '<strong>✓ ' + acc.memberName + '</strong> | A/C: ' + acc.accountNumber +

  // ── Build Transactions Page ──────────────────────────────────
  // manualStartSr = operator-set last printed Sr. No. (global variable)
  // NO header ever — sirf pure data rows
  function buildTxnPage(acc, res, manualStartSr) {
    var ROWS_PER_PAGE = 25;
    var startAfter    = parseInt(manualStartSr) || 0;
    var txns          = res.transactions || [];

    if (txns.length === 0) {
      document.getElementById('pbTxnPage').innerHTML =
        '<div style="padding:20px;text-align:center;color:#27ae60;font-family:sans-serif;">' +
        '\u2705 Koi nayi transaction nahi — passbook up to date hai.</div>';
      return;
    }

    var srNo     = startAfter + 1;
    var allPages = [];
    var pageRows = '';
    var rowCount = 0;

    for (var i = 0; i < txns.length; i++) {
      var t = txns[i];

      if (rowCount > 0 && rowCount >= ROWS_PER_PAGE) {
        allPages.push(_buildPage(pageRows));
        pageRows = '';
        rowCount = 0;
      }

      var isW = (parseFloat(t.amount) < 0) ||
        (t.type && (
          t.type.toUpperCase().indexOf('WITH')  !== -1 ||
          t.type.toUpperCase().indexOf('CLOSE') !== -1 ||
          t.type.toUpperCase().indexOf('DEBIT') !== -1
        ));
      var amt  = Math.abs(parseFloat(t.amount) || 0).toFixed(2);
      var desc = String(t.type || '').replace(/</g,'&lt;').replace(/>/g,'&gt;');
      if (desc.length > 30) desc = desc.substring(0, 29) + '\u2026';

      var d = String(t.date || '');
      if (d.length > 8) {
        var dt = new Date(d);
        if (!isNaN(dt.getTime())) {
          d = ('0'+dt.getDate()).slice(-2) + '-' +
              ('0'+(dt.getMonth()+1)).slice(-2) + '-' +
              String(dt.getFullYear()).slice(-2);
        }
      }

      pageRows +=
        '<tr>' +
          '<td class="pb-col-sr">'   + srNo                                   + '</td>' +
          '<td class="pb-col-date">' + d                                       + '</td>' +
          '<td class="pb-col-desc">' + desc                                    + '</td>' +
          '<td class="pb-col-dep">'  + (!isW ? amt : '')                      + '</td>' +
          '<td class="pb-col-with">' + (isW  ? amt : '')                      + '</td>' +
          '<td class="pb-col-bal">'  + (parseFloat(t.balance)||0).toFixed(2) + '</td>' +
          '<td class="pb-col-init">'                                            + '</td>' +
        '</tr>';
      srNo++;
      rowCount++;
    }

    if (pageRows) allPages.push(_buildPage(pageRows));
    document.getElementById('pbTxnPage').innerHTML = allPages.join('');
  }

  // ── One page — ONLY rows, zero header, zero footer ───────────
  function _buildPage(rowsHtml) {
    return '<div class="pb-page pb-txn-page">' +
      '<table class="pb-txn-table" style="table-layout:fixed;width:100%;border-collapse:collapse;">' +
        '<colgroup>' +
          '<col style="width:22px;">' +
          '<col style="width:58px;">' +
          '<col>' +
          '<col style="width:68px;">' +
          '<col style="width:68px;">' +
          '<col style="width:72px;">' +
          '<col style="width:38px;">' +
        '</colgroup>' +
        '<tbody>' + rowsHtml + '</tbody>' +
      '</table>' +
    '</div>';
  }


  // ── Print ──
  function confirmAndPrint() {
    const label = selectedPrintType === 'NEW' ? 'COVER PAGE' : 'TRANSACTIONS';
    if (!confirm('Print ' + label + ' for ' + selectedAccount.memberName +
                 '?\n\nOperator: ' + currentUser)) return;

    window.print();

    if (selectedPrintType === 'SUBSEQUENT') {
      const txnIds = transactionsData.transactions.map(function(t) { return t.txnId; });
      google.script.run
        .withSuccessHandler(function(res) {
          if (!res) return;
          if (res.success) {
            showMessage('✅ ' + res.message, 'success');
            setTimeout(resetForm, 3000);
          } else {
            showMessage('⚠️ Print done but marking failed: ' + res.message, 'error');
          }
        })
        .withFailureHandler(function(err) {
          showMessage('⚠️ Print done but marking failed: ' + (err.message || err), 'error');
        })
        .markTransactionsAsPrinted(selectedAccount.accountNumber, txnIds, currentUser);
    } else {
      showMessage('✅ Cover page printed by ' + currentUser + '!', 'success');
      setTimeout(resetForm, 2000);
    }
  }
