  <script>
    // ✅ FIXED INIT: Single clean load event — no race conditions, no dead code
    let operatorId    = null;
    let operatorToken = null;
    let userBranchCode = '';

    window.addEventListener('load', function() {
      // Step 1: Get operatorId — URL param first, then window inject fallback
      operatorId    = new URLSearchParams(window.location.search).get('operator') || (window.OPERATOR_ID || null);
      operatorToken = (new URLSearchParams(window.location.search).get('token'))
                      ? decodeURIComponent(new URLSearchParams(window.location.search).get('token'))
                      : (window.OPERATOR_TOKEN || null);

      if (!operatorId) {
        alert('❌ Operator ID not found!\n\nPlease close this window and login again.');
        window.close();
        return;
      }

      // Step 2: Show operator ID in header
      const header = document.getElementById('headerTitle');
      if (header) {
        header.innerHTML += '<br><small style="font-size:14px;opacity:0.9;">Operator: ' + operatorId + '</small>';
      }

      // Step 3: Fetch branch code + show branch banner
      google.script.run
        .withSuccessHandler(function(result) {
          if (result && result.success) {
            userBranchCode = result.branchCode || '';
            const bannerDiv = document.createElement('div');
            bannerDiv.style.cssText = 'background:linear-gradient(135deg,#667eea,#764ba2);color:white;padding:8px 20px;text-align:center;font-size:13px;';
            bannerDiv.innerHTML = '<strong>Operator:</strong> ' + operatorId + ' &nbsp;|&nbsp; <strong>Branch:</strong> ' + userBranchCode;
            const container = document.querySelector('.container') || document.body;
            container.insertBefore(bannerDiv, container.firstChild);
          }
        })
        .getOperatorDetails(operatorId);

      // Step 4: Load form data
      initFormData();
    });

    function initFormData() {
      loadAccounts();
    }

    document.getElementById('depositDate').valueAsDate = new Date();
      document.getElementById('depositDate').readOnly = true; // ✅ No backdate
    const today = new Date();
    document.getElementById('monthFor').value = today.getFullYear() + '-' + String(today.getMonth() + 1).padStart(2, '0');
      document.getElementById('monthFor').readOnly = true; // ✅ No backdate



    function loadAccounts() {
      google.script.run
        .withSuccessHandler(function(res) {
          if (res.success) {
            const select = document.getElementById('accountNo');
            res.accounts.forEach(acc => {
              const option = document.createElement('option');
              option.value = acc.accountNo;
              option.textContent = acc.accountNo + ' - ' + acc.name;
              option.dataset.name = acc.name;
              option.dataset.email = acc.email;
              option.dataset.userId = acc.userId;
              select.appendChild(option);
            });
          }
        })
        .getActiveAccountsForDeposit(operatorId);
    }

    document.getElementById('accountNo').addEventListener('change', function() {
      const selected = this.options[this.selectedIndex];
      if (selected.value) {
        document.getElementById('memberName').value = selected.dataset.name || '';
        document.getElementById('email').value = selected.dataset.email || '';
        document.getElementById('userId').value = selected.dataset.userId || '';
        loadBalance(selected.value);
      }
    });

    function loadBalance(accountNo) {
      google.script.run
        .withSuccessHandler(function(res) {
          if (res.success) {
            document.getElementById('currentBalance').textContent = res.balance.toFixed(2);
            document.getElementById('balanceDisplay').style.display = 'block';
          }
        })
        .getCurrentBalance(accountNo);
    }

    document.getElementById('depositForm').addEventListener('submit', function(e) {
      e.preventDefault();
      submitDeposit();
    });

    function submitDeposit() {
  const form = document.getElementById('depositForm');
  
  if (!form.checkValidity()) {
    form.reportValidity();
    return;
  }

  // ✅ FIXED: Correct field IDs matching the HTML
  const depositData = {
    accountNo:  document.getElementById('accountNo').value.trim(),
    name:       document.getElementById('memberName').value.trim(),
    amount:     parseFloat(document.getElementById('amount').value) || 0,
    interest:   parseFloat(document.getElementById('interest').value) || 0,
    date:       document.getElementById('depositDate').value,
    monthFor:   document.getElementById('monthFor').value,
    email:      document.getElementById('email').value.trim(),
    userId:     document.getElementById('userId').value.trim(),
    branchCode: userBranchCode
  };

  if (!depositData.accountNo || !depositData.amount || !depositData.date || !depositData.monthFor) {
    showError('❌ Please fill all required fields');
    const btn2 = document.getElementById('submitBtn');
    btn2.disabled = false;
    btn2.textContent = '📤 Submit for Supervisor Approval';
    return;
  }

  const btn = document.getElementById('submitBtn');
  btn.disabled = true;
  btn.textContent = '⏳ Submitting...';

  console.log('📤 Submitting deposit | Operator:', operatorId, '| Account:', depositData.accountNo, '| Amount:', depositData.amount);

  google.script.run
    .withSuccessHandler(function(res) {
      btn.disabled = false;
      btn.textContent = '📤 Submit for Supervisor Approval';

      if (res.success) {
        showSuccess(`✅ ${res.message}<br>Entry ID: ${res.entryId}<br>Operator: ${operatorId}`);
        document.getElementById('depositForm').reset();
        document.getElementById('depositDate').valueAsDate = new Date();
      document.getElementById('depositDate').readOnly = true; // ✅ No backdate
        document.getElementById('balanceDisplay').style.display = 'none';
        loadAccounts();
      } else {
        showError('❌ ' + res.message);
      }
    })
    .withFailureHandler(function(err) {
      btn.disabled = false;
      btn.textContent = '📤 Submit for Supervisor Approval';
      showError('❌ Error: ' + err.message);
    })
    // ✅ FIXED: operatorId PEHLE, depositData BAAD MEIN
    .operatorSubmitDeposit(operatorId, depositData, operatorToken);
}
    function showError(msg) {
      document.getElementById('errorMsg').innerHTML = msg;
      document.getElementById('errorMsg').style.display = 'block';
      document.getElementById('successMsg').style.display = 'none';
    }

    function showSuccess(msg) {
      document.getElementById('successMsg').innerHTML = msg;
      document.getElementById('successMsg').style.display = 'block';
      document.getElementById('errorMsg').style.display = 'none';
    }
    
  </script>
