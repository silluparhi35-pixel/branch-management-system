<script>
  var operatorId    = null;
  var operatorToken = null;
  var currentBalance = 0;
  var activeTxnTab   = 'deposit';

  window.addEventListener('load', function() {
    operatorId    = new URLSearchParams(window.location.search).get('operator') || (window.OPERATOR_ID || null);
    operatorToken = (new URLSearchParams(window.location.search).get('token'))
                    ? decodeURIComponent(new URLSearchParams(window.location.search).get('token'))
                    : (window.OPERATOR_TOKEN || null);

    if (!operatorId) {
      alert('❌ Operator ID not found! Please login again.');
      window.close(); return;
    }

    document.getElementById('headerTitle').innerHTML +=
      '<br><small style="font-size:13px;opacity:0.9;">Operator: ' + operatorId + '</small>';

    // Set dates readonly
    var today = new Date();
    document.getElementById('depositDate').valueAsDate    = today;
    document.getElementById('withdrawalDate').valueAsDate = today;

    // Branch banner
    google.script.run
      .withSuccessHandler(function(res) {
        if (res && res.success) {
          var b = document.createElement('div');
          b.style.cssText = 'background:linear-gradient(135deg,#0d7377,#14a085);color:white;padding:8px 20px;text-align:center;font-size:13px;';
          b.innerHTML = '<strong>Operator:</strong> ' + operatorId +
                        ' &nbsp;|&nbsp; <strong>Branch:</strong> ' + (res.branchCode || '');
          var c = document.querySelector('.container');
          c.insertBefore(b, c.firstChild);
        }
      })
      .getOperatorDetails(operatorId);

    loadAccounts();
  });

  // ── Tab switch ──────────────────────────────────────────────
  function switchTxnTab(tab) {
    activeTxnTab = tab;
    document.getElementById('tabDeposit').classList.toggle('active', tab === 'deposit');
    document.getElementById('tabWithdrawal').classList.toggle('active', tab === 'withdrawal');
    document.getElementById('tabDeposit').classList.toggle('deposit', true);
    document.getElementById('tabWithdrawal').classList.toggle('withdrawal', true);
    document.getElementById('depositPanel').classList.toggle('active', tab === 'deposit');
    document.getElementById('withdrawalPanel').classList.toggle('active', tab === 'withdrawal');
    clearMessages();
  }

  // ── Load SAVINGS accounts ───────────────────────────────────
  function loadAccounts() {
    google.script.run
      .withSuccessHandler(function(res) {
        var sel = document.getElementById('accountNo');
        sel.innerHTML = '<option value="">-- Select SAVINGS Account --</option>';
        if (res.success && res.accounts && res.accounts.length > 0) {
          res.accounts.forEach(function(acc) {
            var opt = document.createElement('option');
            opt.value           = acc.accountNo;
            opt.textContent     = acc.accountNo + '  —  ' + acc.name;
            opt.dataset.name    = acc.name;
            opt.dataset.email   = acc.email;
            opt.dataset.userId  = acc.userId;
            sel.appendChild(opt);
          });
        } else {
          sel.innerHTML = '<option value="">No SAVINGS accounts in your branch</option>';
        }
      })
      .withFailureHandler(function(err) {
        document.getElementById('accountNo').innerHTML =
          '<option value="">Error loading accounts</option>';
        showError('❌ ' + err.message);
      })
      .getSavingsAccountsForOperator(operatorId);
  }

  function onAccountChange() {
    var sel = document.getElementById('accountNo');
    var opt = sel.options[sel.selectedIndex];

    if (!opt.value) {
      document.getElementById('memberName').value       = '';
      document.getElementById('balanceCard').style.display = 'none';
      document.getElementById('depEmail').value         = '';
      document.getElementById('depUserId').value        = '';
      document.getElementById('wdEmail').value          = '';
      document.getElementById('wdUserId').value         = '';
      currentBalance = 0;
      document.getElementById('afterDepCard').style.display = 'none';
      document.getElementById('afterWdCard').style.display  = 'none';
      return;
    }

    document.getElementById('memberName').value  = opt.dataset.name   || '';
    document.getElementById('depEmail').value    = opt.dataset.email  || '';
    document.getElementById('depUserId').value   = opt.dataset.userId || '';
    document.getElementById('wdEmail').value     = opt.dataset.email  || '';
    document.getElementById('wdUserId').value    = opt.dataset.userId || '';

    // Load balance
    google.script.run
      .withSuccessHandler(function(res) {
        currentBalance = res.success ? (res.balance || 0) : 0;
        document.getElementById('balanceAmt').textContent = currentBalance.toFixed(2);
        document.getElementById('balanceCard').style.display = 'block';
        calcAfter('deposit');
        calcAfter('withdrawal');
      })
      .getSavingsBalance(opt.value);
  }

  // ── Live balance preview ────────────────────────────────────
  function calcAfter(type) {
    if (type === 'deposit') {
      var amt  = parseFloat(document.getElementById('depositAmount').value) || 0;
      var card = document.getElementById('afterDepCard');
      if (amt > 0) {
        document.getElementById('afterDepAmt').textContent = (currentBalance + amt).toFixed(2);
        card.className = 'after-card positive';
        card.style.display = 'block';
      } else {
        card.style.display = 'none';
      }
    } else {
      var wAmt  = parseFloat(document.getElementById('withdrawalAmount').value) || 0;
      var wCard = document.getElementById('afterWdCard');
      if (wAmt > 0) {
        var remaining = currentBalance - wAmt;
        document.getElementById('afterWdAmt').textContent = remaining.toFixed(2);
        wCard.className = remaining < 0 ? 'after-card negative' : 'after-card positive';
        wCard.style.display = 'block';
      } else {
        wCard.style.display = 'none';
      }
    }
  }

  // ── Submit deposit ──────────────────────────────────────────
  function submitDeposit() {
    var accountNo = document.getElementById('accountNo').value;
    var amount    = parseFloat(document.getElementById('depositAmount').value) || 0;
    var date      = document.getElementById('depositDate').value;

    if (!accountNo) { showError('❌ Select an account'); return; }
    if (!amount || amount <= 0) { showError('❌ Enter a valid deposit amount'); return; }

    var data = {
      accountNo:  accountNo,
      name:       document.getElementById('memberName').value,
      amount:     amount,
      date:       date,
      remarks:    document.getElementById('depositRemarks').value.trim(),
      email:      document.getElementById('depEmail').value,
      userId:     document.getElementById('depUserId').value,
      txnType:    'DEPOSIT'
    };

    var btn = document.getElementById('btnDeposit');
    btn.disabled = true; btn.textContent = '⏳ Submitting...';

    google.script.run
      .withSuccessHandler(function(res) {
        btn.disabled = false;
        btn.textContent = '📤 Submit Deposit for Approval';
        if (res.success) {
          showSuccess('✅ Deposit submitted!<br><strong>Entry ID:</strong> ' + res.entryId +
                      '<br>Supervisor approval ke baad balance update hoga.');
          document.getElementById('depositAmount').value  = '';
          document.getElementById('depositRemarks').value = '';
          document.getElementById('afterDepCard').style.display = 'none';
        } else {
          showError('❌ ' + res.message);
        }
      })
      .withFailureHandler(function(err) {
        btn.disabled = false;
        btn.textContent = '📤 Submit Deposit for Approval';
        showError('❌ ' + err.message);
      })
      .operatorSubmitSavingsDeposit(operatorId, data, operatorToken);
  }

  // ── Submit withdrawal ───────────────────────────────────────
  function submitWithdrawal() {
    var accountNo = document.getElementById('accountNo').value;
    var amount    = parseFloat(document.getElementById('withdrawalAmount').value) || 0;
    var date      = document.getElementById('withdrawalDate').value;

    if (!accountNo) { showError('❌ Select an account'); return; }
    if (!amount || amount <= 0) { showError('❌ Enter a valid withdrawal amount'); return; }
    if (amount > currentBalance) {
      showError('❌ Insufficient balance!\n\nAvailable: ₹' + currentBalance.toFixed(2) +
                '\nRequested: ₹' + amount.toFixed(2));
      return;
    }

    var data = {
      accountNo:  accountNo,
      name:       document.getElementById('memberName').value,
      amount:     amount,
      date:       date,
      remarks:    document.getElementById('withdrawalRemarks').value.trim(),
      email:      document.getElementById('wdEmail').value,
      userId:     document.getElementById('wdUserId').value,
      txnType:    'WITHDRAWAL'
    };

    var btn = document.getElementById('btnWithdrawal');
    btn.disabled = true; btn.textContent = '⏳ Submitting...';

    google.script.run
      .withSuccessHandler(function(res) {
        btn.disabled = false;
        btn.textContent = '📤 Submit Withdrawal for Approval';
        if (res.success) {
          showSuccess('✅ Withdrawal submitted!<br><strong>Entry ID:</strong> ' + res.entryId +
                      '<br>Supervisor approval ke baad amount deducted hoga.');
          document.getElementById('withdrawalAmount').value  = '';
          document.getElementById('withdrawalRemarks').value = '';
          document.getElementById('afterWdCard').style.display = 'none';
        } else {
          showError('❌ ' + res.message);
        }
      })
      .withFailureHandler(function(err) {
        btn.disabled = false;
        btn.textContent = '📤 Submit Withdrawal for Approval';
        showError('❌ ' + err.message);
      })
      .operatorSubmitSavingsWithdrawal(operatorId, data, operatorToken);
  }

  function clearMessages() {
    document.getElementById('errorMsg').style.display  = 'none';
    document.getElementById('successMsg').style.display = 'none';
  }
  function showError(msg) {
    var el = document.getElementById('errorMsg');
    el.textContent = msg; el.style.display = 'block';
    document.getElementById('successMsg').style.display = 'none';
  }
  function showSuccess(msg) {
    var el = document.getElementById('successMsg');
    el.innerHTML = msg; el.style.display = 'block';
    document.getElementById('errorMsg').style.display = 'none';
  }
</script>
