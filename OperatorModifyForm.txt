<script>
  let operatorId    = null;
  let operatorToken = null;
  let currentMemberId = null;
  let accountsData    = [];  // full account list for selected member

  // ── INIT ──────────────────────────────────────────────────────
  window.addEventListener('load', function() {
    operatorId    = new URLSearchParams(window.location.search).get('operator') || (window.OPERATOR_ID || null);
    operatorToken = (new URLSearchParams(window.location.search).get('token'))
                    ? decodeURIComponent(new URLSearchParams(window.location.search).get('token'))
                    : (window.OPERATOR_TOKEN || null);

    if (!operatorId) {
      alert('❌ Operator ID not found! Please login again.');
      window.close(); return;
    }

    document.getElementById('headerTitle').innerHTML +=
      '<br><small style="font-size:13px;opacity:0.9;">Operator: ' + operatorId + '</small>';

    // Branch banner
    google.script.run
      .withSuccessHandler(function(res) {
        if (res && res.success) {
          var b = document.createElement('div');
          b.style.cssText = 'background:linear-gradient(135deg,#667eea,#764ba2);color:white;padding:8px 20px;text-align:center;font-size:13px;';
          b.innerHTML = '<strong>Operator:</strong> ' + operatorId +
                        ' &nbsp;|&nbsp; <strong>Branch:</strong> ' + (res.branchCode || '');
          var c = document.querySelector('.container');
          c.insertBefore(b, c.firstChild);
        }
      })
      .getOperatorDetails(operatorId);

    // Enter key on search
    document.getElementById('searchInput').addEventListener('keydown', function(e) {
      if (e.key === 'Enter') searchMember();
    });
  });

  // ── SEARCH ────────────────────────────────────────────────────
  function searchMember() {
    var term = document.getElementById('searchInput').value.trim();
    if (term.length < 2) {
      document.getElementById('searchResults').innerHTML =
        '<p style="color:#dc3545;font-size:13px;margin-top:8px;">Enter at least 2 characters</p>';
      return;
    }

    document.getElementById('searchResults').innerHTML =
      '<p style="color:#888;font-size:13px;margin-top:8px;">Searching...</p>';

    google.script.run
      .withSuccessHandler(function(res) {
        var container = document.getElementById('searchResults');
        if (!res.success || res.members.length === 0) {
          container.innerHTML = '<p style="color:#dc3545;font-size:13px;margin-top:8px;">No members found</p>';
          return;
        }
        var html = '';
        res.members.forEach(function(m) {
          html += '<div class="result-item" onclick="selectMember(\'' + escHtml(m.memberId) + '\')">' +
            '<div><div class="result-name">' + escHtml(m.name) + '</div>' +
            '<div class="result-id">ID: ' + escHtml(m.memberId) + ' &nbsp;|&nbsp; User: ' + escHtml(m.userId) + '</div></div>' +
            '<button class="btn-select">Select ✓</button></div>';
        });
        container.innerHTML = html;
      })
      .withFailureHandler(function(e) {
        document.getElementById('searchResults').innerHTML =
          '<p style="color:#dc3545;font-size:13px;">Error: ' + e.message + '</p>';
      })
      .searchMemberForModification(term, operatorId);
  }

  function selectMember(memberId) {
    currentMemberId = memberId;
    document.getElementById('searchResults').innerHTML =
      '<p style="color:#27ae60;font-size:13px;margin-top:8px;">⏳ Loading member data...</p>';

    google.script.run
      .withSuccessHandler(function(res) {
        if (!res.success) {
          showError('❌ ' + res.message);
          return;
        }
        populateMemberForm(res.member);
        populateAccountDropdown(res.accounts);
        accountsData = res.accounts;

        document.getElementById('mainForm').style.display = 'block';
        document.getElementById('searchResults').innerHTML =
          '<p style="color:#27ae60;font-size:13px;margin-top:8px;">✅ Member loaded: ' + escHtml(res.member.name) + '</p>';
        document.getElementById('errorMsg').style.display  = 'none';
        document.getElementById('successMsg').style.display = 'none';
      })
      .withFailureHandler(function(e) {
        showError('❌ Error: ' + e.message);
      })
      .getMemberAndAccountForModification(memberId, operatorId);
  }

  // ── POPULATE FORMS ────────────────────────────────────────────
  function populateMemberForm(m) {
    document.getElementById('memberId').value  = m.memberId;
    document.getElementById('mUserId').value   = m.userId;
    document.getElementById('mName').value     = m.name;
    document.getElementById('mContact').value  = m.contact;
    document.getElementById('mEmail').value    = m.email;
    document.getElementById('mNameCurr').textContent    = 'Current: ' + m.name;
    document.getElementById('mContactCurr').textContent = 'Current: ' + m.contact;
    document.getElementById('mEmailCurr').textContent   = 'Current: ' + m.email;
  }

  function populateAccountDropdown(accounts) {
    var sel = document.getElementById('accountSelect');
    sel.innerHTML = '<option value="">-- Select Account --</option>';
    accounts.forEach(function(a) {
      var opt = document.createElement('option');
      opt.value       = a.accountNo;
      opt.textContent = a.accountNo + ' (' + a.accountType + ')';
      sel.appendChild(opt);
    });
    document.getElementById('accountFields').style.display = 'none';
    document.getElementById('noAccountMsg').style.display  = 'block';
  }

  function loadAccountData() {
    var selNo = document.getElementById('accountSelect').value;
    if (!selNo) {
      document.getElementById('accountFields').style.display = 'none';
      document.getElementById('noAccountMsg').style.display  = 'block';
      return;
    }
    var acc = accountsData.find(function(a) { return a.accountNo === selNo; });
    if (!acc) return;

    document.getElementById('aNominee').value   = acc.nomineeName;
    document.getElementById('aRelation').value  = acc.relationship;
    document.getElementById('aEmail').value     = acc.email;
    document.getElementById('aPhotoUrl').value  = acc.photoUrl;

    // DOB format: dd/MM/yyyy → yyyy-MM-dd for input[type=date]
    if (acc.dob) {
      var parts = acc.dob.split('/');
      if (parts.length === 3) {
        document.getElementById('aDob').value = parts[2] + '-' + parts[1] + '-' + parts[0];
      }
    } else {
      document.getElementById('aDob').value = '';
    }

    document.getElementById('aNomineeCurr').textContent   = 'Current: ' + (acc.nomineeName || '—');
    document.getElementById('aRelationCurr').textContent  = 'Current: ' + (acc.relationship || '—');
    document.getElementById('aEmailCurr').textContent     = 'Current: ' + (acc.email || '—');
    document.getElementById('aPhotoUrlCurr').textContent  = acc.photoUrl ? 'Current: Photo set ✓' : 'Current: No photo';
    document.getElementById('aDobCurr').textContent       = 'Current: ' + (acc.dob || '—');

    // Photo preview
    if (acc.photoUrl) {
      var img = document.getElementById('photoImg');
      img.src = acc.photoUrl.replace('view', 'preview');
      document.getElementById('photoPreview').style.display = 'block';
    } else {
      document.getElementById('photoPreview').style.display = 'none';
    }

    // Photo URL change listener
    document.getElementById('aPhotoUrl').oninput = function() {
      var url = this.value.trim();
      if (url) {
        var img = document.getElementById('photoImg');
        // Convert Drive share link to direct preview
        var match = url.match(/\/d\/([a-zA-Z0-9_-]{25,})/);
        if (match) url = 'https://drive.google.com/thumbnail?id=' + match[1] + '&sz=w400';
        img.src = url;
        document.getElementById('photoPreview').style.display = 'block';
      } else {
        document.getElementById('photoPreview').style.display = 'none';
      }
    };

    document.getElementById('accountFields').style.display = 'block';
    document.getElementById('noAccountMsg').style.display  = 'none';
  }

  // ── SUBMIT MEMBER MODIFICATION ────────────────────────────────
  function submitMemberModification() {
    var memberId = document.getElementById('memberId').value.trim();
    if (!memberId) { showError('❌ No member selected'); return; }

    var name    = document.getElementById('mName').value.trim();
    var contact = document.getElementById('mContact').value.trim();
    var email   = document.getElementById('mEmail').value.trim();

    if (!name && !contact && !email) {
      showError('❌ Koi bhi field change nahi ki. Modify karne ke liye kuch update karo.');
      return;
    }
    if (contact && !/^\d{10}$/.test(contact)) {
      showError('❌ Mobile number must be exactly 10 digits'); return;
    }
    if (email && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
      showError('❌ Invalid email format'); return;
    }

    var modData = {
      memberId: memberId,
      userId:   document.getElementById('mUserId').value.trim()
    };
    if (name)    modData.name    = name;
    if (contact) modData.contact = contact;
    if (email)   modData.email   = email;

    var btn = document.getElementById('submitMemberBtn');
    btn.disabled = true;
    btn.textContent = '⏳ Submitting...';

    google.script.run
      .withSuccessHandler(function(res) {
        btn.disabled = false;
        btn.textContent = '📤 Submit Member Changes for Approval';
        if (res.success) {
          showSuccess('✅ ' + res.message + '<br><strong>Entry ID:</strong> ' + res.entryId +
                      '<br><small>Supervisor approval ke baad changes apply honge.</small>');
        } else {
          showError('❌ ' + res.message);
        }
      })
      .withFailureHandler(function(e) {
        btn.disabled = false;
        btn.textContent = '📤 Submit Member Changes for Approval';
        showError('❌ ' + e.message);
      })
      .operatorSubmitMemberModification(operatorId, modData, operatorToken);
  }

  // ── SUBMIT ACCOUNT MODIFICATION ───────────────────────────────
  function submitAccountModification() {
    var accountNo = document.getElementById('accountSelect').value;
    if (!accountNo) { showError('❌ Please select an account'); return; }

    var nominee  = document.getElementById('aNominee').value.trim();
    var relation = document.getElementById('aRelation').value.trim();
    var email    = document.getElementById('aEmail').value.trim();
    var photoUrl = document.getElementById('aPhotoUrl').value.trim();
    var dob      = document.getElementById('aDob').value;

    if (!nominee && !relation && !email && !photoUrl && !dob) {
      showError('❌ Koi bhi field change nahi ki. Modify karne ke liye kuch update karo.');
      return;
    }

    var modData = {
      accountNo: accountNo,
      memberId:  currentMemberId
    };
    if (nominee)  modData.nomineeName  = nominee;
    if (relation) modData.relationship = relation;
    if (email)    modData.email        = email;
    if (photoUrl) modData.photoUrl     = photoUrl;
    if (dob)      modData.dob          = dob;

    var btn = document.getElementById('submitAccountBtn');
    btn.disabled = true;
    btn.textContent = '⏳ Submitting...';

    google.script.run
      .withSuccessHandler(function(res) {
        btn.disabled = false;
        btn.textContent = '📤 Submit Account Changes for Approval';
        if (res.success) {
          showSuccess('✅ ' + res.message + '<br><strong>Entry ID:</strong> ' + res.entryId +
                      '<br><small>Supervisor approval ke baad changes apply honge.</small>');
        } else {
          showError('❌ ' + res.message);
        }
      })
      .withFailureHandler(function(e) {
        btn.disabled = false;
        btn.textContent = '📤 Submit Account Changes for Approval';
        showError('❌ ' + e.message);
      })
      .operatorSubmitAccountModification(operatorId, modData, operatorToken);
  }

  // ── TABS ──────────────────────────────────────────────────────
  function switchTab(tabId, el) {
    document.querySelectorAll('.tab-content').forEach(function(t) { t.classList.remove('active'); });
    document.querySelectorAll('.tab').forEach(function(t) { t.classList.remove('active'); });
    document.getElementById(tabId).classList.add('active');
    el.classList.add('active');
    document.getElementById('errorMsg').style.display  = 'none';
    document.getElementById('successMsg').style.display = 'none';
  }

  function showError(msg) {
    var el = document.getElementById('errorMsg');
    el.innerHTML = msg; el.style.display = 'block';
    document.getElementById('successMsg').style.display = 'none';
    el.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
  }
  function showSuccess(msg) {
    var el = document.getElementById('successMsg');
    el.innerHTML = msg; el.style.display = 'block';
    document.getElementById('errorMsg').style.display = 'none';
    el.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
  }
  function escHtml(s) {
    return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }
