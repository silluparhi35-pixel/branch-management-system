// ── Treasurer Login ───────────────────────
function doTreasurerLogin() {
  var uid   = document.getElementById('trsUserId').value.trim();
  var pwd   = document.getElementById('trsPwd').value.trim();
  var errEl = document.getElementById('trsError');
  var btn   = document.getElementById('trsLoginBtn');

  if (!uid || !pwd) { errEl.textContent = 'Please enter User ID and Password'; errEl.style.display = 'block'; return; }

  btn.disabled = true; btn.textContent = 'Logging in...';
  errEl.style.display = 'none';

  google.script.run
    .withSuccessHandler(function(res) {
      btn.disabled = false; btn.textContent = 'Login';
      if (res && res.success) {
        currentUserId = uid.toUpperCase();
        currentSessionToken = res.token || '';   // ✅ Store session token
        document.getElementById('trsLoginOverlay').style.display = 'none';
        document.getElementById('currentUserDisplay').textContent = currentUserId;
        document.getElementById('topbarBadge').textContent = 'TREASURER';

        if (res.officeId) {
          selectedOfficeId = res.officeId;
          // ✅ After login: only load treasurer's own assigned office
          loadTreasurerOfficeOnly(res.officeId, res.officeName);
        } else {
          // No office assigned — show warning
          document.getElementById('officeSelector').innerHTML =
            '<option value="">⚠️ No office assigned</option>';
          alert('⚠️ No Cash Office is assigned to your account.\nPlease contact Admin to assign an office.');
        }

        // ✅ Load treasurer wallet balance
        loadTreasurerWallet();
        // ✅ DAY LOCK: Login ke baad status check karo
        setTimeout(function() { checkAndApplyDayLock(); }, 300);
        // ✅ Holiday warning — non-blocking banner
        if (res.isHoliday && res.holidayWarning) {
          setTimeout(function() {
            var banner = document.createElement('div');
            banner.style.cssText = 'position:fixed;top:20px;left:50%;transform:translateX(-50%);' +
              'background:#e67e22;color:white;padding:16px 28px;border-radius:12px;' +
              'z-index:9999;font-size:14px;font-weight:bold;text-align:center;' +
              'box-shadow:0 4px 20px rgba(0,0,0,0.4);max-width:90vw;line-height:1.6;';
            banner.innerHTML = res.holidayWarning.replace(/\n/g, '<br>');
            var closeBtn = document.createElement('div');
            closeBtn.innerHTML = '&#10005;';
            closeBtn.style.cssText = 'display:inline-block;margin-left:16px;cursor:pointer;font-size:18px;vertical-align:middle;opacity:0.8;';
            closeBtn.onclick = function() { banner.remove(); };
            banner.appendChild(closeBtn);
            document.body.appendChild(banner);
            setTimeout(function() { if (banner.parentElement) banner.remove(); }, 8000);
          }, 700);
        }

      } else {
        errEl.textContent = res ? res.message : 'Login failed';
        errEl.style.display = 'block';
      }
    })
    .withFailureHandler(function(err) {
      btn.disabled = false; btn.textContent = 'Login';
      errEl.textContent = 'Server error. Try again.';
      errEl.style.display = 'block';
    })
    .treasurerLogin(uid, pwd);
}

// Pages jinpe action hota hai — lock hone chahiye
var LOCKED_PAGES = ['transfer', 'pending'];

// Check day status from backend and apply lock/unlock
function checkAndApplyDayLock() {
  if (!currentUserId) return;
  google.script.run
    .withSuccessHandler(function(res) {
      var status = (res && res.success && res.status) ? res.status : 'NONE';
      applyDayLock(status);
    })
    .withFailureHandler(function() {
      applyDayLock('NONE'); // Safe — lock on failure
    })
    .getTreasurerTodayStatus(currentUserId);
}

// Apply lock state to whole UI
function applyDayLock(status) {
  currentDayStatus = status;

  // ── Status bar update ──
  var bar   = document.getElementById('dayStatusBar');
  var dot   = document.getElementById('dsbDot');
  var label = document.getElementById('dsbLabel');
  var sub   = document.getElementById('dsbSub');
  var today = new Date().toLocaleDateString('en-IN', {day:'2-digit', month:'short', year:'numeric'});

  bar.style.display = 'flex';
  dot.className = 'dsb-dot';

  if (status === 'OPEN') {
    dot.classList.add('open');
    label.textContent = '✅ Day Open';
    label.style.color = 'var(--green)';
    sub.textContent   = today + ' — Transactions allowed';
  } else if (status === 'CLOSED') {
    dot.classList.add('closed');
    label.textContent = '🔒 Day Closed';
    label.style.color = 'var(--red)';
    sub.textContent   = today + ' — Day End ho chuka hai. Kal Day Begin karo.';
  } else {
    dot.classList.add('none');
    label.textContent = '⚠️ Day Begin Nahi Hua';
    label.style.color = 'var(--gold)';
    sub.textContent   = today + ' — Pehle Day Begin karo';
  }

  // ── Nav items lock/unlock ──
  LOCKED_PAGES.forEach(function(pageName) {
    var navEl = document.querySelector('[data-page="' + pageName + '"]');
    if (!navEl) return;
    if (status === 'OPEN') {
      navEl.classList.remove('day-locked');
    } else {
      navEl.classList.add('day-locked');
    }
  });

  // ── Page overlays ──
  LOCKED_PAGES.forEach(function(pageName) {
    _setPageOverlay(pageName, status !== 'OPEN');
  });
}

// Add or remove lock overlay on a specific page
function _setPageOverlay(pageName, locked) {
  var page = document.getElementById('page-' + pageName);
  if (!page) return;

  // Remove existing overlay if any
  var existing = page.querySelector('.page-lock-overlay');
  if (existing) existing.remove();

  if (!locked) return;

  var msg, sub;
  if (currentDayStatus === 'CLOSED') {
    msg = '🔒 Aaj ka Din Band Ho Chuka Hai';
    sub = 'Day End ho chuka hai — kal Day Begin karo';
  } else {
    msg = '⚠️ Day Begin Nahi Hua';
    sub = 'Transactions ke liye pehle Day Begin karo';
  }

  var overlay = document.createElement('div');
  overlay.className = 'page-lock-overlay';
  overlay.innerHTML =
    '<div class="lock-icon">🔐</div>' +
    '<div class="lock-msg">' + msg + '</div>' +
    '<div class="lock-sub">' + sub + '</div>' +
    '<button onclick="showPage(\'dayend\')" style="margin-top:20px;background:var(--gold);color:var(--navy);' +

// Returns true if action is allowed, false otherwise
function isDayOpen(showAlert) {
  if (currentDayStatus === 'OPEN') return true;
  if (showAlert !== false) {
    var msg = currentDayStatus === 'CLOSED'
      ? '🔒 Aaj ka din band ho chuka hai!\n\nDay End ke baad koi transaction nahi ho sakta.\nKal Day Begin karo.'
      : '⚠️ Day Begin nahi hua!\n\nTransaction ke liye pehle Day Begin karo.\n\n"Day Begin / End" section mein jao.';
    alert(msg);
  }
  return false;
}

function doTransfer(direction) {
  // ✅ DAY LOCK — Day Open nahi hai to transfer block
  if (!isDayOpen()) return;

  if (!selectedOfficeId) { showMsg('transferMsg', 'Please select an office first.', 'error'); return; }

  var isBtoO  = (direction === 'BRANCH_TO_OFFICE');
  var branchSel = document.getElementById(isBtoO ? 'txfBranch1' : 'txfBranch2').value;
  var amtRaw    = parseFloat(document.getElementById(isBtoO ? 'txfAmount1' : 'txfAmount2').value);
  var remarks   = document.getElementById(isBtoO ? 'txfRemarks1' : 'txfRemarks2').value.trim();

  if (!branchSel)           { showMsg('transferMsg', 'Branch select karo.', 'error'); return; }
  if (!amtRaw || amtRaw <= 0) { showMsg('transferMsg', 'Valid amount enter karo.', 'error'); return; }

  var branchName = document.getElementById(isBtoO ? 'txfBranch1' : 'txfBranch2').selectedOptions[0].text;
  var dirLabel   = isBtoO ? 'Branch → Office' : 'Office → Branch';

  document.getElementById('confirmTitle').textContent = 'Confirm Transfer';
  document.getElementById('confirmBody').innerHTML =
    '<b>Direction:</b> ' + dirLabel + '<br>' +
    '<b>Branch:</b> ' + branchName + '<br>' +
    '<b>Amount:</b> <span class="mono" style="color:var(--gold);font-size:16px;">' + fmt(amtRaw) + '</span><br>' +
    (remarks ? '<b>Remarks:</b> ' + remarks : '');

  document.getElementById('confirmOkBtn').onclick = function() {
    closeModal('confirmModal');
    execTransfer(direction, branchSel, amtRaw, remarks, isBtoO);
  };
  document.getElementById('confirmModal').style.display = 'flex';
}

function execTransfer(direction, branchCode, amount, remarks, isBtoO) {
  var panel = document.querySelector(isBtoO ? '.dir-in' : '.dir-out').closest('.transfer-panel');
  var btn   = panel.querySelector('.btn');
  btn.disabled = true;
  btn.innerHTML = '<span class="spinner-sm"></span> Processing...';

  var btnLabel = isBtoO ? '↑ Record Receipt' : '↓ Record Dispatch';

  function onSuccess(res) {
    btn.disabled = false;
    btn.innerHTML = btnLabel;
    if (!res) { showMsg('transferMsg', 'No response', 'error'); return; }
    if (res.success) {
      var msg = '✅ ' + res.message;
      if (res.txnId) msg += ' (Txn: ' + res.txnId + ')';
      if (res.newWalletBalance !== undefined) {
        msg += ' | 💰 Wallet: ₹' + res.newWalletBalance.toLocaleString('en-IN', {minimumFractionDigits:2});
      }
      showMsg('transferMsg', msg, 'success');
      document.getElementById(isBtoO ? 'txfAmount1'  : 'txfAmount2').value  = '';
      document.getElementById(isBtoO ? 'txfRemarks1' : 'txfRemarks2').value = '';
      loadDashboard();
      loadTreasurerWallet();
    } else {
      showMsg('transferMsg', '❌ ' + res.message, 'error');
    }
  }

  function onError(e) {
    btn.disabled = false;
    btn.innerHTML = btnLabel;
    showMsg('transferMsg', '❌ Error: ' + e.message, 'error');
  }

  // FIX: Correct argument order for each backend function
  // branchSendCashToOffice(officeId, branchCode, amount, remarks, processedBy)
  // officeSendCashToBranchPending(treasurerId, officeId, branchCode, amount, remarks)
  if (isBtoO) {
    google.script.run
      .withSuccessHandler(onSuccess)
      .withFailureHandler(onError)
      .branchSendCashToOffice(selectedOfficeId, branchCode, amount, remarks, currentUserId);
  } else {
    google.script.run
      .withSuccessHandler(onSuccess)
      .withFailureHandler(onError)
      .officeSendCashToBranchPending(currentUserId, selectedOfficeId, branchCode, amount, remarks);
  }
}

function loadCashBook() {
  var date = document.getElementById('cashBookDate').value;
  if (!date) { showMsg('cashBookMsg', 'Please select a date.', 'error'); return; }

  document.getElementById('cashBookContainer').innerHTML =
    '<div style="text-align:center;padding:40px;color:var(--muted);"><span class="spinner-sm"></span> Generating cash book...</div>';

  google.script.run
    .withSuccessHandler(function(res) {
      if (!res || !res.success) {
        document.getElementById('cashBookContainer').innerHTML =
          '<div class="empty-state"><div class="icon">📒</div><p>' + (res ? res.message : 'Error') + '</p></div>';
        return;
      }
      renderCashBook(res);
    })
    .withFailureHandler(function(e) {
      showMsg('cashBookMsg', '❌ ' + e.message, 'error');
      document.getElementById('cashBookContainer').innerHTML = '';
    })
    .getTreasurerCashBook(currentUserId, date);
}

function renderCashBook(d) {
  var matchColor = d.balanceMatch ? 'var(--green)' : 'var(--red)';
  var matchIcon  = d.balanceMatch ? '✅' : '❌';

  var html =
    '<div id="cashBookPrintArea" style="background:var(--navy2);border:1px solid var(--border);border-radius:12px;padding:24px;">' +

    // ── Header ──
    '<div style="text-align:center;border-bottom:2px solid var(--gold);padding-bottom:14px;margin-bottom:20px;">' +
      '<div style="font-size:11px;color:var(--muted);letter-spacing:2px;text-transform:uppercase;margin-bottom:4px;">Treasurer Cash Book</div>' +
      '<div style="font-size:20px;font-weight:700;color:var(--gold);">' + d.officeId + '</div>' +
      '<div style="font-size:13px;color:var(--muted);margin-top:4px;">Date: <strong style="color:var(--white);">' + d.reportDate + '</strong>' +
      '&nbsp;&nbsp;|&nbsp;&nbsp;Treasurer: <strong style="color:var(--white);">' + d.treasurerId + '</strong></div>' +
    '</div>' +

    // ── Opening Balance ──
    _cbRow('🏦 Opening Balance', d.openingBalance, 'var(--gold)', true) +

    // ── Receipts ──
    '<div style="margin:16px 0 6px;font-size:11px;letter-spacing:2px;color:var(--green);text-transform:uppercase;border-bottom:1px solid var(--green);padding-bottom:4px;">📥 Receipts — Cash from Branches</div>';

  if (d.receipts && d.receipts.length > 0) {
    d.receipts.forEach(function(r) {
      html += _cbDetailRow(
        '↑ ' + escapeHtml(r.branchCode||'') + (r.remarks ? ' — ' + escapeHtml(r.remarks) : ''),
        r.amount, 'var(--green)',
        r.status === 'COMPLETED' ? '✅' : '⏳'
      );
    });
  } else {
    html += '<div style="font-size:12px;color:var(--muted);padding:8px 0;">No receipts for this date</div>';
  }

  html += _cbRow('Total Received', d.totalReceived, 'var(--green)', true) +

    // ── Payments ──
    '<div style="margin:16px 0 6px;font-size:11px;letter-spacing:2px;color:var(--red);text-transform:uppercase;border-bottom:1px solid var(--red);padding-bottom:4px;">📤 Payments — Cash to Branches</div>';

  if (d.payments && d.payments.length > 0) {
    d.payments.forEach(function(p) {
      html += _cbDetailRow(
        '↓ ' + p.branchCode + (p.remarks ? ' — ' + p.remarks : ''),
        p.amount, 'var(--red)',
        p.status === 'COMPLETED' ? '✅' : '⏳ Pending ACK'
      );
    });
  } else {
    html += '<div style="font-size:12px;color:var(--muted);padding:8px 0;">No payments for this date</div>';
  }

  html += _cbRow('Total Sent', d.totalSent, 'var(--red)', true) +

    // ── Closing Balance ──
    '<div style="margin-top:20px;padding:16px;background:rgba(212,168,67,0.08);border:1px solid var(--gold);border-radius:8px;">' +
      '<div style="display:flex;justify-content:space-between;align-items:center;">' +
        '<span style="font-weight:700;font-size:15px;color:var(--gold);">🏁 Closing Balance</span>' +
        '<span style="font-size:22px;font-weight:700;color:var(--gold);">' + fmt(d.closingBalance) + '</span>' +
      '</div>' +
      '<div style="margin-top:12px;padding-top:12px;border-top:1px dashed var(--border);display:flex;justify-content:space-between;align-items:center;">' +
        '<span style="font-size:13px;color:var(--muted);">' + matchIcon + ' Wallet Balance Check</span>' +
        '<span style="font-size:15px;font-weight:700;color:' + matchColor + ';">' + fmt(d.walletBalance) + '</span>' +
      '</div>' +
      (!d.balanceMatch
        ? '<div style="color:var(--red);font-size:12px;margin-top:8px;">⚠️ Difference: ' + fmt(d.difference) + ' — Please verify transactions</div>'
        : '<div style="color:var(--green);font-size:12px;margin-top:8px;">✅ Wallet balance matches Cash Book closing balance</div>'
      ) +
    '</div>' +

    // ── Summary box ──
    '<div style="display:grid;grid-template-columns:repeat(3,1fr);gap:12px;margin-top:16px;">' +
      _cbStat('Opening', d.openingBalance, 'var(--gold)') +
      _cbStat('+ Received', d.totalReceived, 'var(--green)') +
      _cbStat('− Sent', d.totalSent, 'var(--red)') +
    '</div>' +

    '</div>'; // /cashBookPrintArea

  document.getElementById('cashBookContainer').innerHTML = html;
}

function loadDayEndStatus() {
  var box = document.getElementById('dayendStatusBox');
  if (!box) return;
  box.innerHTML = '<span style="color:var(--muted);">Checking...</span>';

  google.script.run
    .withSuccessHandler(function(res) {
      if (!res || !res.success) {
        box.innerHTML = '<span style="color:var(--red);">Could not load status. Retry karo.</span>';
        return;
      }
      var today  = new Date().toLocaleDateString('en-IN', {day:'2-digit',month:'short',year:'numeric'});
      var status = res.status || null;
      var color  = status === 'CLOSED' ? 'var(--red)'   : status === 'OPEN' ? 'var(--green)' : 'var(--gold)';
      var icon   = status === 'CLOSED' ? '🔒'           : status === 'OPEN' ? '✅'           : '⚠️';
      var lbl    = status === 'CLOSED' ? 'Day CLOSED'   : status === 'OPEN' ? 'Day OPEN'     : 'Day Begin Nahi Hua';

      box.innerHTML =
        '<div style="font-size:36px;margin-bottom:10px;">' + icon + '</div>' +
        '<div style="font-size:16px;font-weight:700;color:' + color + ';margin-bottom:6px;">' + lbl + '</div>' +
        '<div style="font-size:12px;color:var(--muted);">Date: ' + today + '</div>' +
        (res.openingBalance !== undefined
          ? '<div style="margin-top:10px;font-size:13px;color:var(--text);">Opening: <strong style="color:var(--gold);">' + fmt(res.openingBalance) + '</strong>' +
            (res.status === 'CLOSED' ? '&nbsp;&nbsp;|&nbsp;&nbsp;Closing: <strong style="color:var(--green);">' + fmt(res.closingBalance) + '</strong>' : '') + '</div>'
          : '<div style="margin-top:8px;font-size:12px;color:var(--muted);">Aaj ka koi record nahi — Day Begin karo</div>'
        );

      // ✅ FIX: applyDayLock sirf tab call karo jab status clearly milta ho
      // 'NONE' ya null pe override mat karo — Day Begin ke baad OPEN state protect karo
      if (status === 'OPEN' || status === 'CLOSED') {
        applyDayLock(status);
      }
      // Agar status null/undefined hai (no record yet) aur currentDayStatus already OPEN/CLOSED
      // hai toh override mat karo — woh state valid hai
      else if (currentDayStatus === 'NONE') {
        applyDayLock('NONE');
      }

      loadDayEndLog();
    })
    .withFailureHandler(function(e) {
      box.innerHTML = '<span style="color:var(--red);">Error: ' + escapeHtml(e.message||'') + '</span>';
    })
    .getTreasurerTodayStatus(currentUserId);
}

function doDayBegin() {
  if (!currentUserId) { alert('Please login first.'); return; }
  if (!confirm('Day Begin karo?\n\nAaj ka opening balance permanently save hoga.\nSab transactions enable ho jaayenge.')) return;

  var btn = document.getElementById('dayBeginBtn');
  btn.disabled = true; btn.textContent = '⏳ Saving...';

  google.script.run
    .withSuccessHandler(function(res) {
      btn.disabled = false; btn.textContent = '📂 Day Begin';
      var el = document.getElementById('dayBeginResult');
      el.style.display = 'block';
      if (res && res.success) {
        el.style.color = 'var(--green)';
        el.innerHTML   = res.message.replace(/\n/g, '<br>');
        // ✅ FIX: status field se lock set karo
        // alreadySaved case mein status 'OPEN' ya 'CLOSED' aata hai
        // fresh Day Begin mein status field nahi hota — default OPEN
        var lockStatus = res.status || 'OPEN';
        applyDayLock(lockStatus);
        loadDayEndLog();
      } else {
        el.style.color = 'var(--red)';
        el.textContent = res ? res.message : 'Error occurred';
      }
    })
    .withFailureHandler(function(e) {
      btn.disabled = false; btn.textContent = '📂 Day Begin';
      showMsg('dayendMsg', '❌ ' + e.message, 'error');
    })
    .saveTreasurerOpeningBalance(currentUserId);
}

function doDayEnd() {
  if (!currentUserId) { alert('Please login first.'); return; }
  if (currentDayStatus !== 'OPEN') {
    alert('⚠️ Day Begin pehle karo!\n\nDay End tabhi ho sakta hai jab Day Begin hua ho.');
    return;
  }
  if (!confirm('Day End karo?\n\nAaj ka closing balance permanently save hoga.\nSab transactions band ho jaayenge.\n\nYeh action confirm karo.')) return;

  var btn = document.getElementById('dayEndBtn');
  btn.disabled = true; btn.textContent = '⏳ Saving...';

  google.script.run
    .withSuccessHandler(function(res) {
      btn.disabled = false; btn.textContent = '🔒 Day End';
      var el = document.getElementById('dayEndResult');
      el.style.display = 'block';
      if (res && res.success) {
        el.style.color = 'var(--green)';
        el.innerHTML   = res.message.replace(/\n/g, '<br>');
        // ✅ Lock all transactions immediately
        applyDayLock('CLOSED');
        loadDayEndStatus();
        loadDashboard();
      } else {
        el.style.color = 'var(--red)';
        el.textContent = res ? res.message : 'Error occurred';
      }
    })
    .withFailureHandler(function(e) {
      btn.disabled = false; btn.textContent = '🔒 Day End';
      showMsg('dayendMsg', '❌ ' + e.message, 'error');
    })
    .saveTreasurerClosingBalance(currentUserId);
}

function loadDayEndLog() {
  var tbody = document.getElementById('dayendLogBody');
  if (!tbody) return;
  tbody.innerHTML = '<tr class="loading-row"><td colspan="4"><span class="spinner-sm"></span> Loading...</td></tr>';

  google.script.run
    .withSuccessHandler(function(res) {
      if (!res || !res.success || !res.records || res.records.length === 0) {
        tbody.innerHTML = '<tr class="loading-row"><td colspan="4">Koi record nahi mila</td></tr>';
        return;
      }
      var rows = '';
      res.records.forEach(function(r) {
        var sc = r.status === 'CLOSED' ? 'var(--green)' : 'var(--gold)';
        rows +=
          '<tr>' +
          '<td>' + r.date + '</td>' +
          '<td style="text-align:right;color:var(--gold);">' + fmt(r.opening) + '</td>' +
          '<td style="text-align:right;color:' + (r.status === 'CLOSED' ? 'var(--green)' : 'var(--muted)') + ';">' +
            (r.status === 'CLOSED' ? fmt(r.closing) : '—') + '</td>' +
          '<td style="text-align:center;"><span style="color:' + sc + ';font-size:11px;font-weight:700;">' + r.status + '</span></td>' +
          '</tr>';
      });
      tbody.innerHTML = rows;
    })
    .withFailureHandler(function(e) {
      tbody.innerHTML = '<tr class="loading-row"><td colspan="4">Error: ' + escapeHtml(e.message||'') + '</td></tr>';
    })
    .getTreasurerBalanceLog(currentUserId);
}

function openTreasurerChangePassword() {
  var oldP = prompt('Enter CURRENT password:');
  if (!oldP) return;
  var newP = prompt('Enter NEW password (min 4 chars):');
  if (!newP || newP.length < 4) { alert('❌ Password too short!'); return; }
  var cnf  = prompt('CONFIRM new password:');
  if (newP !== cnf) { alert('❌ Passwords do not match!'); return; }

  google.script.run
    .withSuccessHandler(function(res) {
      if (res.success) {
        alert('✅ Password changed successfully!\n\nPlease login again with new password.');
        location.reload();
      } else {
        alert('❌ ' + res.message);
      }
    })
    .withFailureHandler(function(e) { alert('❌ Error: ' + e.message); })
    .changePasswordWithOldVerify(currentUserId, oldP, newP, currentSessionToken);
}
