// Global variables
let currentUserId = null;
let editingAnnouncementId = null;

// ==========================================
// LOGIN HANDLING
// ==========================================
function handleLogin(event) {
  event.preventDefault();
  
  const userId = document.getElementById('userId').value.trim();
  const password = document.getElementById('password').value.trim();
  const errorDiv = document.getElementById('loginError');
  const loginBtn = document.getElementById('loginBtn');
  
  if (!userId || !password) {
    showError('Please enter both User ID and Password');
    return;
  }
  
  // Disable button and show loading
  loginBtn.disabled = true;
  loginBtn.textContent = 'Logging in...';
  errorDiv.classList.remove('show');
  
  google.script.run
    .withSuccessHandler(function(result) {
      loginBtn.disabled = false;
      loginBtn.textContent = 'Login';
      
      if (result.success) {
        // Store session
        currentUserId = result.adminId || result.userId;
        
        // Hide login page, show dashboard
        document.getElementById('loginPage').style.display = 'none';
        document.getElementById('dashboardPage').style.display = 'block';
        document.getElementById('logoutBtn').style.display = 'block';
        
        // Load announcements
        loadAnnouncements();
        loadBranches();
        loadHolidays();
      } else {
        showError(result.message || 'Login failed');
      }
    })
    .withFailureHandler(function(error) {
      loginBtn.disabled = false;
      loginBtn.textContent = 'Login';
      showError('Error: ' + error.message);
    })
    .adminLogin(userId, password);
}

function showError(message) {
  const errorDiv = document.getElementById('loginError');
  errorDiv.textContent = '❌ ' + message;
  errorDiv.classList.add('show');
  
  // Auto-hide after 5 seconds
  setTimeout(function() {
    errorDiv.classList.remove('show');
  }, 5000);
}

// ==========================================
// LOAD ANNOUNCEMENTS
// ==========================================
function loadAnnouncements() {
  const listDiv = document.getElementById('announcementsList');
  listDiv.innerHTML = '<div class="loading"><div class="spinner"></div>Loading announcements...</div>';
  
  google.script.run
    .withSuccessHandler(function(result) {
      if (result.success) {
        displayAnnouncements(result.announcements || []);
      } else {
        listDiv.innerHTML = '<div class="empty-state"><div class="empty-state-icon">❌</div><p>Error: ' + result.message + '</p></div>';
      }
    })
    .withFailureHandler(function(error) {
      listDiv.innerHTML = '<div class="empty-state"><div class="empty-state-icon">❌</div><p>Error: ' + error.message + '</p></div>';
    })
    .getAllAnnouncementsForAdmin();
}

// ==========================================
// DISPLAY ANNOUNCEMENTS
// ==========================================
function displayAnnouncements(announcements) {
  const listDiv = document.getElementById('announcementsList');
  
  if (announcements.length === 0) {
    listDiv.innerHTML = `
      <div class="empty-state">
        <div class="empty-state-icon">📢</div>
        <p>No announcements yet. Create your first announcement!</p>
      </div>
    `;
    return;
  }
  
  let html = '';
  
  announcements.forEach(ann => {
    const target = getTargetLabel(ann.target, ann.branchCode);
    
    html += `
      <div class="announcement-item">
        <div class="announcement-header">
          <div>
            <div class="announcement-title">${escapeHtml(ann.title || 'Untitled')}</div>
            <div class="announcement-meta">
              Created: ${escapeHtml(ann.createdDate || 'N/A')} by ${escapeHtml(ann.createdBy || 'Admin')}
            </div>
          </div>
        </div>
        
        <div class="announcement-target">
          🎯 ${escapeHtml(target)}
        </div>
        
        <div class="announcement-message">
          ${escapeHtml(ann.message || '')}
        </div>
        
        <div class="announcement-actions">
          <button onclick="editAnnouncement('${escapeHtml(ann.id || '')}', '${escapeHtml(ann.title)}', '${escapeHtml(ann.message)}', '${escapeHtml(ann.target || '')}', '${escapeHtml(ann.branchCode || '')}')" 
                  class="btn-success">
            ✏️ Edit
          </button>
          <button onclick="deleteAnnouncement('${escapeHtml(ann.id || '')}')" 
                  class="btn-danger">
            🗑️ Delete
          </button>
        </div>
      </div>
    `;
  });
  
  listDiv.innerHTML = html;
}

// ==========================================
// HELPER FUNCTIONS
// ==========================================
function getTargetLabel(target, branchCode) {
  switch(target) {
    case 'ALL': return 'All Users';
    case 'ALL_STAFF': return 'All Staff (Operators & Supervisors)';
    case 'ALL_OPERATORS': return 'All Operators';
    case 'ALL_SUPERVISORS': return 'All Supervisors';
    case 'BRANCH_SPECIFIC': return 'Branch: ' + (branchCode || 'N/A') + ' (All)';
    case 'BRANCH_STAFF': return 'Branch: ' + (branchCode || 'N/A') + ' (Staff)';
    case 'BRANCH_MEMBERS': return 'Branch: ' + (branchCode || 'N/A') + ' (Members)';
    default: return target;
  }
}

function escapeHtml(text) {
  if (!text) return '';
  return String(text).replace(/'/g, "\\'").replace(/"/g, '&quot;');
}

// ==========================================
// MODAL FUNCTIONS
// ==========================================
function openCreateModal() {
  editingAnnouncementId = null;
  document.getElementById('modalTitle').textContent = '📢 Create Announcement';
  document.getElementById('announcementTitle').value = '';
  document.getElementById('announcementMessage').value = '';
  document.getElementById('announcementTarget').value = 'ALL';
  document.getElementById('branchSelectGroup').style.display = 'none';
  
  document.getElementById('announcementModal').classList.add('active');
}

function editAnnouncement(id, title, message, target, branchCode) {
  editingAnnouncementId = id;
  document.getElementById('modalTitle').textContent = '✏️ Edit Announcement';
  document.getElementById('announcementTitle').value = title;
  document.getElementById('announcementMessage').value = message;
  document.getElementById('announcementTarget').value = target;
  
  if (target === 'BRANCH_SPECIFIC' || target === 'BRANCH_STAFF' || target === 'BRANCH_MEMBERS') {
    document.getElementById('branchSelectGroup').style.display = 'block';
    document.getElementById('announcementBranch').value = branchCode;
  } else {
    document.getElementById('branchSelectGroup').style.display = 'none';
  }
  
  document.getElementById('announcementModal').classList.add('active');
}

function closeModal() {
  document.getElementById('announcementModal').classList.remove('active');
  editingAnnouncementId = null;
}

// ==========================================
// SAVE ANNOUNCEMENT
// ==========================================
function saveAnnouncement() {
  const title = document.getElementById('announcementTitle').value.trim();
  const message = document.getElementById('announcementMessage').value.trim();
  const target = document.getElementById('announcementTarget').value;
  const branchCode = document.getElementById('announcementBranch').value;
  
  if (!title) {
    alert('⚠️ Please enter a title');
    return;
  }
  
  if (!message) {
    alert('⚠️ Please enter a message');
    return;
  }
  
  if ((target === 'BRANCH_SPECIFIC' || target === 'BRANCH_STAFF' || target === 'BRANCH_MEMBERS') && !branchCode) {
    alert('⚠️ Please select a branch');
    return;
  }
  
  const data = {
    title: title,
    message: message,
    target: target,
    branchCode: branchCode || '',
    createdBy: currentUserId
  };
  
  if (editingAnnouncementId) {
    // Update existing
    google.script.run
      .withSuccessHandler(function(result) {
        if (result.success) {
          alert('✅ Announcement updated successfully!');
          closeModal();
          loadAnnouncements();
        } else {
          alert('❌ Error: ' + result.message);
        }
      })
      .withFailureHandler(function(error) {
        alert('❌ Error: ' + error.message);
      })
      .updateAnnouncementAdmin(editingAnnouncementId, data);
  } else {
    // Create new
    google.script.run
      .withSuccessHandler(function(result) {
        if (result.success) {
          alert('✅ Announcement created successfully!');
          closeModal();
          loadAnnouncements();
        } else {
          alert('❌ Error: ' + result.message);
        }
      })
      .withFailureHandler(function(error) {
        alert('❌ Error: ' + error.message);
      })
      .createAnnouncementAdmin(data);
  }
}

// ==========================================
// LOAD BRANCHES
// ==========================================
function loadBranches() {
  const select = document.getElementById('announcementBranch');
  
  google.script.run
    .withSuccessHandler(function(result) {
      if (result.success && result.branches) {
        let options = '<option value="">-- Select Branch --</option>';
        result.branches.forEach(branch => {
          options += `<option value="${branch.code}">${branch.code} - ${branch.name || branch.code}</option>`;
        });
        select.innerHTML = options;
      } else {
        select.innerHTML = '<option value="">-- No branches found --</option>';
      }
    })
    .withFailureHandler(function(error) {
      select.innerHTML = '<option value="">-- Error loading branches --</option>';
    })
    .getAllBranches();
}

// ==========================================
// HOLIDAY MANAGEMENT
// ==========================================
function loadHolidays() {
  const listDiv = document.getElementById('holidayList');
  listDiv.innerHTML = '<div class="loading"><div class="spinner"></div>Loading holidays...</div>';

  google.script.run
    .withSuccessHandler(function(result) {
      if (result.success) {
        displayHolidays(result.holidays || []);
      } else {
        listDiv.innerHTML = '<div class="empty-state"><div class="empty-state-icon">❌</div><p>Error: ' + result.message + '</p></div>';
      }
    })
    .withFailureHandler(function(error) {
      listDiv.innerHTML = '<div class="empty-state"><div class="empty-state-icon">❌</div><p>Error: ' + error.message + '</p></div>';
    })
    .getHolidayList(currentUserId);
}

function displayHolidays(holidays) {
  const listDiv = document.getElementById('holidayList');

  if (holidays.length === 0) {
    listDiv.innerHTML = `
      <div class="empty-state">
        <div class="empty-state-icon">📅</div>
        <p>No holidays added yet. Add your first holiday above!</p>
      </div>
    `;
    return;
  }

  // Sort by date descending for display
  let html = '';
  holidays.forEach(function(h) {
    html += `
      <div class="holiday-item">
        <div class="holiday-info">
          <div class="holiday-date-badge">📅 ${escapeHtml(h.date || '')}</div>
          <div>
            <div class="holiday-desc">${escapeHtml(h.description || 'Office Holiday')}</div>
            <span class="holiday-block-badge">🔒 Login Blocked</span>
          </div>
        </div>
        <button onclick="deleteHolidayEntry(${h.rowIndex})" class="btn-danger" style="padding: 8px 16px; font-size: 13px; white-space: nowrap;">
          🗑️ Delete
        </button>
      </div>
    `;
  });

  listDiv.innerHTML = html;
}

function addHolidayEntry() {
  const dateInput = document.getElementById('holidayDate').value;
  const desc      = document.getElementById('holidayDesc').value.trim();

  if (!dateInput) {
    alert('⚠️ Please select a date');
    return;
  }

  // Convert from yyyy-MM-dd (date picker) → dd/MM/yyyy (backend format)
  const [yyyy, mm, dd] = dateInput.split('-');
  const dateStr = dd + '/' + mm + '/' + yyyy;

  google.script.run
    .withSuccessHandler(function(result) {
      if (result.success) {
        alert('✅ ' + result.message);
        document.getElementById('holidayDate').value = '';
        document.getElementById('holidayDesc').value = '';
        loadHolidays();
      } else {
        alert('❌ ' + result.message);
      }
    })
    .withFailureHandler(function(error) {
      alert('❌ Error: ' + error.message);
    })
    .addHoliday(dateStr, desc, currentUserId);
}

function deleteHolidayEntry(rowIndex) {
  if (!confirm('⚠️ Are you sure you want to delete this holiday?\n\nThis action cannot be undone.')) {
    return;
  }

  google.script.run
    .withSuccessHandler(function(result) {
      if (result.success) {
        alert('✅ Holiday deleted successfully!');
        loadHolidays();
      } else {
        alert('❌ ' + result.message);
      }
    })
    .withFailureHandler(function(error) {
      alert('❌ Error: ' + error.message);
    })
    .deleteHoliday(rowIndex, currentUserId);
}
